const express = require('express'); const multer = require('multer'); const path = require('path'); const fs = require('fs'); const AdmZip = require('adm-zip'); const { v4: uuidv4 } = require('uuid'); const bcrypt = require('bcryptjs'); const cors = require('cors'); const app = express(); const PORT = process.env.PORT || 3001; const ADMIN_PASS = process.env.ADMIN_PASS || 'bas2024'; // Change this! const UPLOADS = path.join(__dirname, 'uploads'); const DATA = path.join(__dirname, 'data.json'); if (!fs.existsSync(UPLOADS)) fs.mkdirSync(UPLOADS, { recursive: true }); // ── DATA ─────────────────────────────────────────────────────────────────────── function load() { if (!fs.existsSync(DATA)) return { modpacks: [], news: [] }; try { return JSON.parse(fs.readFileSync(DATA, 'utf8')); } catch { return { modpacks: [], news: [] }; } } function save(d) { fs.writeFileSync(DATA, JSON.stringify(d, null, 2)); } // ── MIDDLEWARE ───────────────────────────────────────────────────────────────── app.use(cors()); app.use(express.json()); app.use(express.static(path.join(__dirname, 'public'))); // Admin auth middleware function adminAuth(req, res, next) { const token = req.headers['x-admin-token'] || req.query.token; if (token === ADMIN_PASS) return next(); res.status(401).json({ error: 'No autorizado' }); } // ── MULTER ───────────────────────────────────────────────────────────────────── const storage = multer.diskStorage({ destination: UPLOADS, filename: (req, file, cb) => cb(null, `${uuidv4()}.mrpack`) }); const upload = multer({ storage, limits: { fileSize: 1024 * 1024 * 1024 } }); // 1GB // ── API: MODPACKS ───────────────────────────────────────────────────────────── // GET /api/modpacks — list (strip password hash before sending) app.get('/api/modpacks', (req, res) => { const { modpacks } = load(); res.json({ success: true, modpacks: modpacks.map(mp => ({ ...mp, passwordHash: undefined })) }); }); // GET /api/modpacks/:id app.get('/api/modpacks/:id', (req, res) => { const { modpacks } = load(); const mp = modpacks.find(m => m.id === req.params.id); if (!mp) return res.status(404).json({ error: 'No encontrado' }); res.json({ success: true, modpack: { ...mp, passwordHash: undefined } }); }); // POST /api/modpacks/:id/unlock — check password app.post('/api/modpacks/:id/unlock', (req, res) => { const { modpacks } = load(); const mp = modpacks.find(m => m.id === req.params.id); if (!mp) return res.status(404).json({ error: 'No encontrado' }); if (!mp.passwordHash) return res.json({ success: true }); // no password needed const ok = bcrypt.compareSync(req.body.password || '', mp.passwordHash); if (!ok) return res.status(403).json({ success: false, error: 'Contraseña incorrecta' }); res.json({ success: true }); }); // GET /api/modpacks/:id/download app.get('/api/modpacks/:id/download', async (req, res) => { const { modpacks } = load(); const mp = modpacks.find(m => m.id === req.params.id); if (!mp) return res.status(404).json({ error: 'No encontrado' }); // Check password if needed if (mp.passwordHash) { const pass = req.query.password || req.headers['x-pack-password'] || ''; const ok = bcrypt.compareSync(pass, mp.passwordHash); if (!ok) return res.status(403).json({ error: 'Contraseña incorrecta' }); } const file = path.join(UPLOADS, mp.filename); if (!fs.existsSync(file)) return res.status(404).json({ error: 'Archivo no encontrado' }); res.download(file, `${mp.name}.mrpack`); }); // POST /api/modpacks/upload (admin) app.post('/api/modpacks/upload', adminAuth, upload.single('file'), async (req, res) => { try { if (!req.file) return res.status(400).json({ error: 'Sin archivo' }); const name = req.body.name || path.basename(req.file.originalname, '.mrpack'); const password = req.body.password?.trim() || null; const hidden = req.body.hidden === 'true'; let mcVersion = '?', loader = 'Fabric', modsCount = 0, iconB64 = null; try { const zip = new AdmZip(req.file.path); const idx = zip.getEntry('modrinth.index.json'); if (idx) { const d = JSON.parse(idx.getData().toString('utf8')); mcVersion = d.dependencies?.minecraft || '?'; if (d.dependencies?.['fabric-loader']) loader = 'Fabric'; else if (d.dependencies?.forge) loader = 'Forge'; else if (d.dependencies?.['quilt-loader']) loader = 'Quilt'; else if (d.dependencies?.neoforge) loader = 'NeoForge'; modsCount = d.files?.length || 0; } const icon = zip.getEntry('icon.png') || zip.getEntry('pack.png'); if (icon) iconB64 = `data:image/png;base64,${icon.getData().toString('base64')}`; } catch (e) { console.warn('Parse warning:', e.message); } const mp = { id: uuidv4(), name, filename: req.file.filename, mcVersion, loader, modsCount, icon: iconB64, hidden: !!hidden, password: !!password, passwordHash: password ? bcrypt.hashSync(password, 10) : null, size: req.file.size, uploadedAt: new Date().toISOString() }; const data = load(); data.modpacks.push(mp); save(data); res.json({ success: true, modpack: { ...mp, passwordHash: undefined } }); } catch (err) { res.status(500).json({ error: err.message }); } }); // PATCH /api/modpacks/:id (admin) — edit name, password, hidden app.patch('/api/modpacks/:id', adminAuth, async (req, res) => { const data = load(); const idx = data.modpacks.findIndex(m => m.id === req.params.id); if (idx === -1) return res.status(404).json({ error: 'No encontrado' }); const mp = data.modpacks[idx]; if (req.body.name) mp.name = req.body.name; if (req.body.hidden !== undefined) mp.hidden = !!req.body.hidden; if (req.body.password !== undefined) { if (req.body.password) { mp.password = true; mp.passwordHash = bcrypt.hashSync(req.body.password, 10); } else { mp.password = false; mp.passwordHash = null; } } save(data); res.json({ success: true, modpack: { ...mp, passwordHash: undefined } }); }); // DELETE /api/modpacks/:id (admin) app.delete('/api/modpacks/:id', adminAuth, (req, res) => { const data = load(); const mp = data.modpacks.find(m => m.id === req.params.id); if (!mp) return res.status(404).json({ error: 'No encontrado' }); const f = path.join(UPLOADS, mp.filename); if (fs.existsSync(f)) fs.unlinkSync(f); data.modpacks = data.modpacks.filter(m => m.id !== req.params.id); save(data); res.json({ success: true }); }); // ── API: NEWS ────────────────────────────────────────────────────────────────── app.get('/api/news', (req, res) => { const { news } = load(); res.json({ success: true, news: news || [] }); }); app.post('/api/news', adminAuth, (req, res) => { const { title, body, color } = req.body; if (!title) return res.status(400).json({ error: 'Falta título' }); const data = load(); if (!data.news) data.news = []; const item = { id: uuidv4(), title, body: body || '', color: color || 'blue', date: new Date().toISOString() }; data.news.unshift(item); save(data); res.json({ success: true, item }); }); app.delete('/api/news/:id', adminAuth, (req, res) => { const data = load(); data.news = (data.news || []).filter(n => n.id !== req.params.id); save(data); res.json({ success: true }); }); // ── API: ADMIN AUTH CHECK ────────────────────────────────────────────────────── app.post('/api/admin/login', (req, res) => { if (req.body.password === ADMIN_PASS) res.json({ success: true }); else res.status(401).json({ success: false, error: 'Contraseña incorrecta' }); }); // ── START ───────────────────────────────────────────────────────────────────── app.listen(PORT, () => { console.log(`\n🚀 BAS CLIENT Server → http://localhost:${PORT}`); console.log(`🔑 Admin panel → http://localhost:${PORT}/admin.html`); console.log(`🔐 Admin password → ${ADMIN_PASS}\n`); console.log(' Cambia la contraseña con: ADMIN_PASS=tupass node server/index.js\n'); });
